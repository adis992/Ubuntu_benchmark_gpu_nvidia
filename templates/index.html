📦 Installation Help

🖥️ Install NVIDIA Drivers

sudo ubuntu-drivers autoinstall
# Or specific version:
sudo apt install nvidia-driver-565
# After install:
sudo reboot

⚡ Install CUDA Toolkit

wget https://developer.download.nvidia.com/compute/cuda/repos/ubuntu2404/x86_64/cuda-keyring_1.1-1_all.deb
sudo dpkg -i cuda-keyring_1.1-1_all.deb
sudo apt-get update
sudo apt-get -y install cuda-toolkit-12-8

🚀 Install CuPy (GPU Stress)

sudo /opt/nvidia-gpu-benchmark/venv/bin/pip install cupy-cuda12x
# Then restart service:
sudo ./restart_service.sh

🔄 Restart Service

sudo ./restart_service.sh
# Check status:
sudo systemctl status nvidia-gpu-benchmark
# View logs:
sudo journalctl -u nvidia-gpu-benchmark -f

🔍 Diagnostics

nvidia-smi
nvidia-smi -l 2  # Monitor every 2s
nvcc --version
python3 -c "import cupy; cupy.show_config()"

🩺 Check GPU Errors

nvidia-smi --query-gpu=gpu_uuid,ecc.errors.corrected.volatile.total,ecc.errors.uncorrected.volatile.total --format=csv
# Stress test logs:
tail -f /tmp/gpu_stress_0.log
